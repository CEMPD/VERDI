127  protected void setupImageParameters() {
128 
129  // Set basic image layout
130  minX = minY = 0;
131  width = (int) dir.getFieldAsLong(XTIFF.TIFFTAG_IMAGE_WIDTH);
132  height = (int) dir.getFieldAsLong(XTIFF.TIFFTAG_IMAGE_LENGTH);
133

135 
136  // Read the TIFFTAG_BITS_PER_SAMPLE field
137  XTIFFField bitsField = dir.getField(XTIFF.TIFFTAG_BITS_PER_SAMPLE);
138 
139  if (bitsField == null) {
140  // Default
141  bitsPerSample = new char[1];
142  bitsPerSample[0] = 1;
143  } else {
144  bitsPerSample = bitsField.getAsChars();
145  }
146 
147  for (int i = 1; i < bitsPerSample.length; i++) {
148  if (bitsPerSample[i] != bitsPerSample[1]) {
149  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder19"));
150  }
151  }
152 
153  // Get the number of samples per pixel
154  XTIFFField sfield = dir.getField(XTIFF.TIFFTAG_SAMPLES_PER_PIXEL);
155  if (sfield == null) {
156  samplesPerPixel = 1;
157  } else {

163  if (efield == null) {

165  } else {

167  }
168 
169  // Read the TIFFTAG_SAMPLE_FORMAT tag to see whether the data might be
170  // signed or floating point
171  XTIFFField sampleFormatField = dir.getField(XTIFF.TIFFTAG_SAMPLE_FORMAT);
172 
173  if (sampleFormatField != null) {
174  sampleFormat = sampleFormatField.getAsChars();
175 
176  // Check that all the samples have the same format
177  for (int l = 1; l < sampleFormat.length; l++) {
178  if (sampleFormat[l] != sampleFormat[0]) {
179  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder20"));
180  }
181  }
182 
183  } else {
184  sampleFormat = new char[] { 1 };
185  }
186 
187  if (sampleFormat[0] == 1 || sampleFormat[0] == 4) {
188 
189  // Unsigned or unknown
190  if (bitsPerSample[0] == 8) {
191  dataType = DataBuffer.TYPE_BYTE;
192  } else if (bitsPerSample[0] == 16) {
193  dataType = DataBuffer.TYPE_USHORT;
194  } else if (bitsPerSample[0] == 32) {
195  dataType = DataBuffer.TYPE_INT;
196  }
197 
198  } else if (sampleFormat[0] == 2) {
199  // Signed
200 
201  if (bitsPerSample[0] == 1 || bitsPerSample[0] == 4
202  || bitsPerSample[0] == 8) {
203 
204  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder21"));
205 
206  } else if (bitsPerSample[0] == 16) {
207  dataType = DataBuffer.TYPE_SHORT;
208  } else if (bitsPerSample[0] == 32) {
209  dataType = DataBuffer.TYPE_INT;
210  }
211 
212  } else if (sampleFormat[0] == 3) {
213  // Floating point
214  // dataType = DataBuffer.TYPE_FLOAT;
215  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder22"));
216  }
217 
218  if (dir.getField(XTIFF.TIFFTAG_TILE_WIDTH) != null) {
219  // Image is in tiled format
220  tileWidth = (int) dir.getFieldAsLong(XTIFF.TIFFTAG_TILE_WIDTH);
221  tileHeight = (int) dir.getFieldAsLong(XTIFF.TIFFTAG_TILE_LENGTH);
222  tileOffsets = (dir.getField(XTIFF.TIFFTAG_TILE_OFFSETS)).getAsLongs();
223  tileByteCounts = dir.getField(XTIFF.TIFFTAG_TILE_BYTE_COUNTS)
224  .getAsLongs();
225 
226  } else {
227 
228  // Image is in stripped format, looks like tiles to us
229  tileWidth = width;
230  XTIFFField field = dir.getField(XTIFF.TIFFTAG_ROWS_PER_STRIP);
231  if (field == null) {
232  // Default is infinity (2^32 -1), basically the entire image
233  // TODO: Can do a better job of tiling here
234  tileHeight = height;
235  } else {
236  long l = field.getAsLong(0);
237  long infinity = 1;
238  infinity = (infinity << 32) - 1;
239  if (l == infinity) {
240  // 2^32 - 1 (effectively infinity, entire image is 1 strip)
241  tileHeight = height;
242  } else {
243  tileHeight = (int) l;
244  }
245  }
246 
247  XTIFFField tileOffsetsField = dir.getField(XTIFF.TIFFTAG_STRIP_OFFSETS);
248  if (tileOffsetsField == null) {
249  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder11"));
250  } else {
251  tileOffsets = tileOffsetsField.getAsLongs();
252  }
253 
254  XTIFFField tileByteCountsField = dir.getField(XTIFF.TIFFTAG_STRIP_BYTE_COUNTS);
255  if (tileByteCountsField == null) {
256  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder12"));
257  } else {
258  tileByteCounts = tileByteCountsField.getAsLongs();
259  }
260  }

346  } else {

380  // RGB full color image

382

384  tileWidth,
385  tileHeight,
386  bands);
387  } else if (bitsPerSample[0] == 16) {
388

390  tileWidth,
391  tileHeight,
392  bands);

395  }

399 
400  } else if (samplesPerPixel == 3) {
401 
402  image_type = XTIFF.TYPE_RGB;
403  // No alpha
404  colorModel = ImageCodec.createComponentColorModel(sampleModel);
405 
406  } else if (samplesPerPixel == 4) {
407 
408  if (extraSamples == 0) {
409

411  // Transparency.OPAQUE signifies image data that is
412  // completely opaque, meaning that all pixels have an alpha
413  // value of 1.0. So the extra band gets ignored, which is

428

432  // The extra sample here is unassociated alpha, usually a
433  // transparency mask, also called soft matte.
434  colorModel = createAlphaComponentColorModel(dataType,
435  true,
436  false,
437  Transparency.BITMASK);

442

444  // point in trying to reformat the data to be BGR followed by
445  // the ExtraSamples, the way Java2D would like it, because
446  // Java2D can't display it anyway. Therefore create a sample
447  // model with increasing bandOffsets, and keep the colorModel

464  * tileWidth, bandOffsets);
465  colorModel = null;
466  }
467  }

597  public synchronized Raster getTile(int tileX, int tileY) {
598  if (tileX == 0 && tileY == 0 && tile00 != null) {
599  return tile00;
600  }
601 
602  if ((tileX < 0) || (tileX >= tilesX) || (tileY < 0)
603  || (tileY >= tilesY)) {
604  throw new IllegalArgumentException(JaiI18N.getString("XTIFFImageDecoder5"));
605  }
606 
607  // file setup1
608 
609  // Save original file pointer position and seek to tile data location.
610  long save_offset = 0;
611  try {
612  save_offset = stream.getFilePointer();
613  stream.seek(tileOffsets[tileY * tilesX + tileX]);
614  } catch (IOException ioe) {
615  throw new RuntimeException(JaiI18N.getString("XTIFFImageDecoder8"));
616  }
617 
618  // Number of bytes in this tile (strip) after compression.
619  int byteCount = (int) tileByteCounts[tileY * tilesX + tileX];
620 
621  // Find out the number of bytes in the current tile
622  Rectangle tileRect = new Rectangle(tileXToX(tileX), tileYToY(tileY), tileWidth, tileHeight);
623  Rectangle newRect = tileRect.intersection(getBounds());
624 
625  // file setup2
626 
627  byte data[] = new byte[byteCount];
628  WritableRaster tile = null;
629  try {
630  stream.readFully(data, 0, byteCount);
631  tile = codec.decode(this, newRect, data);
632  stream.seek(save_offset);
633  } catch (IOException e) {
634  throw new RuntimeException("Failed to read raw tile data:" + e);
635  }
636 
637  if (tileX == 0 && tileY == 0) {
638  tile00 = tile;
639  }
640  return tile;
641  }
