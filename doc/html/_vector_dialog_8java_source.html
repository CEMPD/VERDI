172  private void findAxisFormulas() {
173  items.clear();
174  DefaultListModel model = (DefaultListModel) formulaList.getModel();
175  xElement = null;
176  yElement = null;
177  tileElement = null;
178  for (int i = 0; i < model.size(); i++) {
179  FormulaListElement element = (FormulaListElement) model.getElementAt(i);
180  if (element.getFormula().equals(fldX.getText())) {
181  xElement = element;
182  }
183 
184  if (element.getFormula().equals(fldY.getText())) {
185  yElement = element;
186  }
187 
188  if (element.getFormula().equals(fldTile.getText())) {
189  tileElement = element;
190  }
191  }
192  if (xElement != null) items.add(xElement);
193  if (yElement != null) items.add(yElement);
194  if (tileElement != null) items.add(tileElement);
195  }
