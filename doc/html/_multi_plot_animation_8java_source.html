234  private void processListSelection() {
235  badIndices.clear();
236  int[] vals = plotList.getCheckBoxListSelectedIndices();
237  Set<Integer> selectedSet = new HashSet<Integer>();
238  for (int val : vals) {
239  selectedSet.add(val);
240  }
241  if (!selectedSet.containsAll(currentSelections)) currentRange = null;
242  currentSelections = selectedSet;
243  if (vals.length == 0) currentRange = null;
244  else {
245  ListModel listModel = plotList.getModel();
246  if (currentRange == null || vals.length == 1) {
247  PlotData plotData = (PlotData) listModel.getElementAt(vals[0]);
248  currentRange = plotData.getRange();
249  currentSteps = plotData.getStepCount(currentRange);
250  }
251  for (int val : vals) {
252  PlotData plotData = (PlotData) listModel.getElementAt(val);
253  DateRange range = plotData.getRange();
254  DateRange newRange = currentRange.overlap(range);
255  if (newRange == null) {
256  badIndices.add(val);
257  } else {
258  int steps = plotData.getStepCount(newRange);
259  if (steps == -1 || steps != currentSteps) {
260  badIndices.add(val);
261  } else {
262  currentRange = newRange;
263  }
264  }
265  }
266  }
