147  /* ARW Users Guide p 3-19
148  <pre>
149 7. MAP_PROJ_NAME: Character string specifying type of map projection. Valid entries are:
150  "polar" -> Polar stereographic
151  "lambert" -> Lambert conformal (secant and tangent)
152  "mercator" -> Mercator
153 
154 8. MOAD_KNOWN_LAT/MOAD_KNOWN_LON (= CEN_LAT, CEN_LON):
155  Real latitude and longitude of the center point in the grid.
156 
157 9. MOAD_STAND_LATS (= TRUE_LAT1/2):
158  2 real values for the "true" latitudes (where grid spacing is exact).
159  Must be between -90 and 90, and the values selected depend on projection:
160  Polar-stereographic: First value must be the latitude at which the grid
161 spacing is true. Most users will set this equal to their center latitude.
162 Second value must be +/-90. for NH/SH grids.
163  Lambert Conformal: Both values should have the same sign as the center
164 latitude. For a tangential lambert conformal, set both to the same value
165 (often equal to the center latitude). For a secant Lambert Conformal, they
166 may be set to different values.
167  Mercator: The first value should be set to the latitude you wish your grid
168  spacing to be true (often your center latitude). Second value is not used.
169 
170 10. MOAD_STAND_LONS (=STAND_LON): This is one entry specifying the longitude in degrees East (-
171 180->180) that is parallel to the y-axis of your grid, (sometimes referred to as the
172 orientation of the grid). This should be set equal to the center longitude in most cases.
173 
174 ---------------------------------
175 version 3.1
176 http://www.mmm.ucar.edu/wrf/users/docs/user_guide_V3.1/users_guide_chap5.htm
177 
178 The definition for map projection options:
179 
180 map_proj = 1: Lambert Conformal
181  2: Polar Stereographic
182  3: Mercator
183  6: latitude and longitude (including global)
184 */

371  private Array convertToDegrees(Variable v) {
372  Array data;
373  try {
374  data = v.read();
375  data = data.reduce();
376  } catch (IOException ioe) {
377  throw new RuntimeException("data read failed on " + v.getFullName() + "=" + ioe.getMessage());
378  }
379  IndexIterator ii = data.getIndexIterator();
380  while (ii.hasNext()) {
381  ii.setDoubleCurrent(Math.toDegrees(ii.getDoubleNext()));
382  }
383  return data;
384  }

541  private CoordinateAxis makeZCoordAxis(NetcdfDataset ds, String axisName, Dimension dim) {
542  if (dim == null) return null;
543 
544  String fromWhere = axisName.endsWith("stag") ? "ZNW" : "ZNU";
545 
546  CoordinateAxis v = new CoordinateAxis1D(ds, null, axisName, DataType.DOUBLE, dim.getShortName(), "", "eta values from variable " + fromWhere);
547  v.addAttribute(new Attribute(CF.POSITIVE, CF.POSITIVE_DOWN)); // eta coordinate is 1.0 at bottom, 0 at top

549  if (!axisName.equals(dim.getShortName()))
550  v.addAttribute(new Attribute(_Coordinate.AliasForDimension, dim.getShortName()));
551

553  // But they are a function of time though the values are the same in the sample file
554  // NOTE: Use first time sample assuming all are the same!
555  Variable etaVar = ds.findVariable(fromWhere);
556  if (etaVar == null) return makeFakeCoordAxis(ds, axisName, dim);
557

559  int[] origin = new int[]{0, 0};
