106  config.putObject(TilePlotConfiguration.COLOR_MAP, colorMapPanel.getColorMap());
107  }
108 
109  int index = tabbedPanel.indexOfTab("Titles");
110 
111  if (index >= 0)
112  config = titlesPanel.fillConfiguration(config);
113 
114  index = tabbedPanel.indexOfTab("Labels");
115 
116  if (index >= 0)
117  config = labelsPanel.fillConfiguration(config);
118 
119  index = tabbedPanel.indexOfTab("Other");
120 
121  if (index >= 0)
122  config = otherPanel.fillConfiguration(config);
123 
124  index = tabbedPanel.indexOfTab("Overlays");
125 
126  if (index >= 0)
127  config = overlays.fillConfiguration(config);
128 
129  plot.configure(config, Plot.ConfigSource.GUI);
130  }
