395  case XTIFFField.TIFF_ASCII:

397  stream.readFully(bvalues, 0, count);
398

400 
401  // Can be multiple strings
402  int index = 0, prevIndex = 0;
403  Vector v = new Vector();
404 
405  while (index < count) {
406 
407  while ((index < count) && (bvalues[index++] != 0))
408  ;
409 
410  // When we encountered zero, means one string has ended
411  v.add(new String(bvalues, prevIndex, (index - prevIndex)));
412  prevIndex = index;
413  }
414 
415  count = v.size();

417  for (int c = 0; c < count; c++) {
418  strings[c] = (String) v.elementAt(c);

424  }

505  private XTIFFField readField() throws IOException,
506  ArrayIndexOutOfBoundsException {
507 // int j;
508  int tag = readUnsignedShort(stream);
509  int type = readUnsignedShort(stream);
510  int count = (int) readUnsignedInt(stream);
511  int value = 0;
512 
513  // The place to return to to read the next tag
514  long nextTagOffset = stream.getFilePointer() + 4;
515 
516  try {
517  // If the tag data can't fit in 4 bytes, the next 4 bytes
518  // contain the starting offset of the data
519  if (count * sizeOfType(type) > 4) {
520  value = (int) (readUnsignedInt(stream));
521  stream.seek(value);
522  }
523  } catch (ArrayIndexOutOfBoundsException ae) {
524  System.err.println(tag + " " + JaiI18N.getString("XTIFFDirectory4"));
525  // if the data type is unknown we should skip this TIFF Field
526  stream.seek(nextTagOffset);
527  throw ae;
528  }
529 
530  Object obj = readFieldValue(tag, type, count);
531 
532  // Position stream at next field and return this one
533  stream.seek(nextTagOffset);
534 
535  return createField(tag, type, count, obj);
536  }
