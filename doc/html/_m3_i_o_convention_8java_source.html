103  protected void constructCoordAxes(NetcdfDataset ds) {
104 
105  Dimension dimx = ds.findDimension("COL");
106  int nx = dimx.getLength();
107 
108  Dimension dimy = ds.findDimension("ROW");
109  int ny = dimy.getLength();
110 
111  int projType = findAttributeInt(ds, "GDTYP");
112  boolean isLatLon = (projType == 1);
113  if (isLatLon) {
114  //ds.addCoordinateAxis( makeCoordAxis( ds, "lon", "COL", nx, "XORIG", "XCELL", "degrees east"));
115  //ds.addCoordinateAxis( makeCoordAxis( ds, "lat", "ROW", ny, "YORIG", "YCELL", "degrees north"));
116 
117  ds.addCoordinateAxis(makeCoordLLAxis(ds, "lon", "COL", nx, "XORIG", "XCELL", "degrees east"));
118  ds.addCoordinateAxis(makeCoordLLAxis(ds, "lat", "ROW", ny, "YORIG", "YCELL", "degrees north"));
119  ct = makeLatLongProjection(ds);
120 
121  VariableDS v = makeCoordinateTransformVariable(ds, ct);
122  ds.addVariable(null, v);
123  v.addAttribute(new Attribute(_Coordinate.Axes, "lon lat"));
124 
125  } else {
126  ds.addCoordinateAxis(makeCoordAxis(ds, "x", "COL", nx, "XORIG", "XCELL", "km"));
127  ds.addCoordinateAxis(makeCoordAxis(ds, "y", "ROW", ny, "YORIG", "YCELL", "km"));
128 
129  if (projType == 2)
130  ct = makeLCProjection(ds);
131  else if (projType == 3)

143  else if (projType == 9)
144  ct = makeAlbersProjection(ds);
145  else if (projType == 10)
146  ct = makeLambertAzimuthalProjection(ds);
147 
148  if (ct != null) {
149  VariableDS v = makeCoordinateTransformVariable(ds, ct);

183  private void makeZCoordAxis(NetcdfDataset ds, String dimName, String layersName, String unitName) {
184  Dimension dimz = ds.findDimension(dimName);
185  int nz = dimz.getLength();
186  ArrayDouble.D1 dataLev = new ArrayDouble.D1(nz);
187  ArrayDouble.D1 dataLayers = new ArrayDouble.D1(nz + 1);
188 
189  // layer values are a numeric global attribute array !!
190  Attribute layers = ds.findGlobalAttribute("VGLVLS");
191  for (int i = 0; i <= nz; i++)
192  dataLayers.set(i, layers.getNumericValue(i).doubleValue());
193 
194  for (int i = 0; i < nz; i++) {
195  double midpoint = (dataLayers.get(i) + dataLayers.get(i + 1)) / 2;
196  dataLev.set(i, midpoint);
197  }
198 
199  CoordinateAxis v = new CoordinateAxis1D(ds, null, "level", DataType.DOUBLE, dimName, unitName,
200  "synthesized coordinate from " + layersName + " global attributes");
201  v.setCachedData(dataLev, true);
202  v.addAttribute(new Attribute("positive", "down"));
203  v.addAttribute(new Attribute(_Coordinate.AxisType, AxisType.GeoZ.toString()));
204 
205  // layer edges
206  String edge_name = "layer";
207  Dimension lay_edge = new Dimension(edge_name, nz + 1);
208 
209  ds.addDimension(null, lay_edge);
210  CoordinateAxis vedge = new CoordinateAxis1D(ds, null, edge_name, DataType.DOUBLE, edge_name, unitName,
211  "synthesized coordinate from " + layersName + " global attributes");
212  vedge.setCachedData(dataLayers, true);
213  v.setBoundaryRef(edge_name);
214  ds.addCoordinateAxis(v);
215  ds.addCoordinateAxis(vedge);
216  }

218  private void makeTimeCoordAxis(NetcdfDataset ds, String timeName) {
219  int start_date = findAttributeInt(ds, "SDATE");
220  int start_time = findAttributeInt(ds, "STIME");
221  int time_step = findAttributeInt(ds, "TSTEP");
222 
223  int year = start_date / 1000;
224  int doy = start_date % 1000;
225  int hour = start_time / 10000;
226  start_time = start_time % 10000;
227  int min = start_time / 100;
228  int sec = start_time % 100;
229 
230  Calendar cal = new GregorianCalendar(new SimpleTimeZone(0, "GMT"));
231  cal.clear();
232  cal.set(Calendar.YEAR, year);
233  cal.set(Calendar.DAY_OF_YEAR, doy);
234  cal.set(Calendar.HOUR_OF_DAY, hour);
235  cal.set(Calendar.MINUTE, min);
236  cal.set(Calendar.SECOND, sec);
237  //cal.setTimeZone( new SimpleTimeZone(0, "GMT"));
238 
239  java.text.SimpleDateFormat dateFormatOut = new java.text.SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
240  dateFormatOut.setTimeZone(java.util.TimeZone.getTimeZone("GMT"));
241 
242  String units = "seconds since " + dateFormatOut.format(cal.getTime()) + " UTC";
243 
244  // parse the time step
245  hour = time_step / 10000;
246  time_step = time_step % 10000;
247  min = time_step / 100;
248  sec = time_step % 100;
249  time_step = hour * 3600 + min * 60 + sec;
250 
251  Dimension dimt = ds.findDimension(timeName);
252  int nt = dimt.getLength();
253  ArrayInt.D1 data = new ArrayInt.D1(nt);
254  for (int i = 0; i < nt; i++) {
255  data.set(i, i * time_step);
256  }
257 
258  // create the coord axis
259  CoordinateAxis1D timeCoord = new CoordinateAxis1D(ds, null, "time", DataType.INT, timeName, units,
260  "synthesized time coordinate from SDATE, STIME, STEP global attributes");
261  timeCoord.setCachedData(data, true);
262  timeCoord.addAttribute(new Attribute(_Coordinate.AxisType, AxisType.Time.toString()));
263 
264  ds.addCoordinateAxis(timeCoord);
265  }
