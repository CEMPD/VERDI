195  public double getScaleFactor(double lat_ts, boolean north) {
196  double e = 0.081819191;
197  double tf = 1.0, mf = 1.0, k0 = 1.0;
198  double root = (1 + e * Math.sin(lat_ts)) / (1 - e * Math.sin(lat_ts));
199  double power = e / 2;
200 
201  if (north)
202  tf = Math.tan(Math.PI / 4 - lat_ts / 2) * (Math.pow(root, power));
203  else
204  tf = Math.tan(Math.PI / 4 + lat_ts / 2) / (Math.pow(root, power));
205 
206  mf = Math.cos(lat_ts) / Math.sqrt(1 - e * e * Math.pow(Math.sin(lat_ts), 2));
207  k0 = mf * Math.sqrt(Math.pow(1 + e, 1 + e) * Math.pow(1 - e, 1 - e)) / (2 * tf);
208 
209  return Double.isNaN(k0) ? 1.0 : k0;
210  }
