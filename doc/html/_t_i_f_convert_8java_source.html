118  protected static void convert(BufferedImage[] image, Object os, int dpi, int compression, float quality) throws IOException {
119 
120  TIFFImageWriter writer = new TIFFImageWriter(new TIFFImageWriterSpi());
121  TIFFImageWriteParam iwp = new TIFFImageWriteParam(writer.getLocale());
122 
123  writer.setOutput(os);
124  writer.prepareWriteSequence(null);
125 
126  iwp.setCompressionMode(TIFFImageWriteParam.MODE_EXPLICIT);
127  iwp.setCompressionType(getCompression(compression));
128 
129  if (!isCompressionLossless(compression)) {
130  if (quality < 0.0F || quality > 1.0F)
131  quality = DEFAULT_COMPRESSION_QUALITY;
132  iwp.setCompressionQuality(quality);
133  }
134 
135  for (int i = 0; i < image.length; i++) {
136  ImageTypeSpecifier imageType = ImageTypeSpecifier.createFromRenderedImage(image[i]);
137  TIFFImageMetadata imageMetadata = (TIFFImageMetadata)writer.getDefaultImageMetadata(imageType, iwp);
138  imageMetadata = createImageMetadata(imageMetadata, image[i].getHeight(), image[i].getWidth(), dpi, compression, image[i].getType());
139  writer.writeToSequence(new IIOImage(image[i], null, imageMetadata), iwp);
140  }
141 
142  writer.endWriteSequence();
143  writer.dispose();
144  }
