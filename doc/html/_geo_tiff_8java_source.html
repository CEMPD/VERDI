131  private void writeGeoKeys() {
132  if (geokeys.size() == 0) return;
133 
134  // count extras
135  int extra_chars = 0;
136  int extra_ints = 0;
137  int extra_doubles = 0;
138  for (GeoKey geokey : geokeys) {
139  if (geokey.isDouble)
140  extra_doubles += geokey.count();
141  else if (geokey.isString)
142  extra_chars += geokey.valueString().length() + 1;
143  else if (geokey.count() > 1)
144  extra_ints += geokey.count();
145  }
146 
147  int n = (geokeys.size()+1) * 4;
148  int[] values = new int[n + extra_ints];
149  double[] dvalues = new double[extra_doubles];
150  char[] cvalues = new char[extra_chars];
151  int icounter = n;
152  int dcounter = 0;
153  int ccounter = 0;
154 
155  values[0] = 1;
156  values[1] = 1;
157  values[2] = 0;
158  values[3] = geokeys.size();
159  int count = 4;
160  for (GeoKey geokey : geokeys) {
161  values[count++] = geokey.tagCode();
162 
163  if (geokey.isDouble) {
164  values[count++] = Tag.GeoDoubleParamsTag.getCode(); // extra double values here
165  values[count++] = geokey.count();
166  values[count++] = dcounter;
167  for (int k=0; k<geokey.count(); k++)
168  dvalues[dcounter++] = geokey.valueD(k);
169 
170  } else if (geokey.isString) {
171  String s = geokey.valueString();
172  values[count++] = Tag.GeoAsciiParamsTag.getCode(); // extra double values here
173  values[count++] = s.length(); // don't include trailing 0 in the count
174  values[count++] = ccounter;
175  for (int k=0; k<s.length(); k++)
176  cvalues[ccounter++] = s.charAt(k);
177  cvalues[ccounter++] = (char) 0;
178 
179  } else if (geokey.count() > 1) { // more than one int value
180  values[count++] = Tag.GeoKeyDirectoryTag.getCode(); // extra int values here
181  values[count++] = geokey.count();
182  values[count++] = icounter;
183  for (int k=0; k<geokey.count(); k++)
184  values[icounter++] = geokey.value(k);
185 
186  } else { // normal case of one int value
187  values[count++] = 0;
188  values[count++] = 1;
189  values[count++] = geokey.value();
190  }
191  } // loop over geokeys
192 
193  addTag(new IFDEntry(Tag.GeoKeyDirectoryTag, FieldType.SHORT).setValue(values));
194  if (extra_doubles > 0)
195  addTag(new IFDEntry(Tag.GeoDoubleParamsTag, FieldType.DOUBLE).setValue(dvalues));
196  if (extra_chars > 0)
197  addTag(new IFDEntry(Tag.GeoAsciiParamsTag, FieldType.ASCII).setValue(new String(cvalues)));
198  }

323  private void writeIFDEntry(FileChannel channel, IFDEntry ifd, int start) throws IOException {
324  channel.position(start);
325  ByteBuffer buffer = ByteBuffer.allocate(12);
326 
327  buffer.putShort( (short) ifd.tag.getCode());
328  buffer.putShort( (short) ifd.type.code);
329  buffer.putInt(ifd.count);
330 
331  int size = ifd.count * ifd.type.size;
332  if (size <= 4) {
333  int done = writeValues(buffer, ifd);
334  for (int k=0; k < 4 - done; k++) // fill out to 4 bytes
335  buffer.put((byte) 0);
336  buffer.flip();
337  channel.write(buffer);
338 
339  } else { // write offset
340  buffer.putInt(nextOverflowData);
341  buffer.flip();
342  channel.write(buffer);
343  // write data
344  channel.position(nextOverflowData);
345  //Logger.debug(" write offset = "+ifd.tag.getName());
346  ByteBuffer vbuffer = ByteBuffer.allocate(size);
347  writeValues(vbuffer, ifd);
348  vbuffer.flip();
349  channel.write(vbuffer);
350  nextOverflowData += size;
351  }
