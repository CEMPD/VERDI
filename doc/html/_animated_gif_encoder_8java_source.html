309  protected int findClosest(Color c) {

312  int g = c.getGreen();
313  int b = c.getBlue();
314  int minpos = 0;

316  int len = colorTab.length;
317  for (int i = 0; i < len;) {

319  int dg = g - (colorTab[i++] & 0xff);
