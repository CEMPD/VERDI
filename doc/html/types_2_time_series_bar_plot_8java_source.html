366  // (int) screenRect.getY()));

375  ChartEntity obj = collection.getEntity(i);

379  if (bounds.getX() >= x) break;

384  int start = frame.getAxes().getTimeStep(dateFormat.parse(entity.getColumnKey().toString()));

387  // (int) (screenRect.getY() + screenRect.getHeight())));
388  x = (int) ((screenRect.getX() + screenRect.getWidth() - insets.left) / panel.getScaleX());
389  for (int i = collection.getEntityCount() - 1; i >= 0; i--) {
390  ChartEntity obj = collection.getEntity(i);
391  if (obj instanceof CategoryItemEntity) {
392  entity = (CategoryItemEntity) obj;
393  Rectangle2D bounds = entity.getArea().getBounds2D();
394  if (bounds.getX() + bounds.getWidth() <= x) break;

396  }

398  // CategoryItemEntity.getCategory() deprecated; replace by getColumnKey(), returns Comparable
399  int end = frame.getAxes().getTimeStep(dateFormat.parse(entity.getColumnKey().toString()));
