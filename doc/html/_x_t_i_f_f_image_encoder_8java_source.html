171  private void computeIFDOffset() {
172

174  * numBands);

601  private void writeDirectory(XTIFFField fields[], int nextIFDOffset)
602  throws IOException {
603 
604  if (currentOffset % 2 == 1) {
605  output.write(0);
606  currentOffset++;
607  }
608 
609  // 2 byte count of number of directory entries (fields)
610  int numEntries = fields.length;
611 
612  long offsetBeyondIFD = currentOffset + 12 * numEntries + 4 + 2;
613  Vector tooBig = new Vector();
614 
615  XTIFFField field;
616  int tag;
617  int type;
618  int count;
619 
620  // Write number of fields in the IFD
621  writeUnsignedShort(numEntries);
622 
623  for (int i = 0; i < numEntries; i++) {
624 
625  field = fields[i];
626 
627  // 12 byte field entry TIFFField
628 
629  // byte 0-1 Tag that identifies a field
630  tag = field.getTag();
631  writeUnsignedShort(tag);
632 
633  // byte 2-3 The field type
634  type = field.getType();
635  writeUnsignedShort(type);
636

638  count = field.getCount();
639  writeLong(count);
640

642  if (count * sizeOfType[type] > 4) {
643 
644  // We need an offset as data won't fit into 4 bytes
645  writeLong(offsetBeyondIFD);
646  offsetBeyondIFD += (count * sizeOfType[type]);
647  tooBig.add(new Integer(i));
648 
649  } else {
650 
651  writeValuesAsFourBytes(field);
652  }
653 
654  }

657  writeLong(nextIFDOffset);
658

661

695  output.write(bytes[i]);
696  }

699  output.write(0);

730  private void writeValues(XTIFFField field) throws IOException {
731 
732  int dataType = field.getType();
733  int count = field.getCount();
734 
735  switch (dataType) {
736 
737  // character data with NULL termination
738  case TIFFField.TIFF_ASCII:
739  String strings[] = field.getAsStrings();
740  for (int i = 0; i < strings.length; i++) {
741  byte bytes[] = strings[i].getBytes();
742  for (int j = 0; j < bytes.length; j++) {
743  output.write(bytes[j]);
744  }
745  if ((i + 1) < count)
746  output.write(0);
747  }
748  break;
749 
750  // unsigned 8 bits
751  case TIFFField.TIFF_BYTE:
752  byte bytes[] = field.getAsBytes();
753  for (int i = 0; i < count; i++) {
754  output.write(bytes[i]);
755  }
756  break;
757 
758  // unsigned 16 bits
759  case TIFFField.TIFF_SHORT:
760  char shorts[] = field.getAsChars();
761  for (int i = 0; i < count; i++) {
762  writeUnsignedShort((int) shorts[i]);
763  }
764  break;
765 
766  // unsigned 32 bits
767  case TIFFField.TIFF_LONG:
768  long longs[] = field.getAsLongs();
769  for (int i = 0; i < count; i++) {
770  writeLong(longs[i]);
771  }
772  break;
773 
774  // IEEE 8-byte double
775  case TIFFField.TIFF_DOUBLE:
776  double doubles[] = field.getAsDoubles();
777  for (int i = 0; i < count; i++) {
778  writeDouble(doubles[i]);
779  }
780  break;
781 
782  case TIFFField.TIFF_RATIONAL:
783  long rationals[][] = field.getAsRationals();
784  for (int i = 0; i < count; i++) {
785  writeLong(rationals[i][0]);
786  writeLong(rationals[i][1]);
787  }
788  break;
789 
790  }

792  }

797  output.write((s & 0xff00) >>> 8);

801  private void writeLong(long l) throws IOException {
802  output.write((int) ((l & 0xff000000) >>> 24));
803  output.write((int) ((l & 0x00ff0000) >>> 16));
804  output.write((int) ((l & 0x0000ff00) >>> 8));
805  output.write(((int) l & 0x000000ff));
806  }
