70  private ValidationResult checkTime(List<AxisRange> ranges) throws IllegalFormulaException {

82  DateRange currRange = new DateRange(axes.getDate(timeAxis.getRange().getOrigin()),
83  axes.getDate((int) endIndex));
84  for (int i = 1; i < variables.size(); i++) {
85  FormulaVariable var = variables.get(i);
86  if (!var.getDataset().equals(formulaVariable.getDataset())) {

91  axes.getDate((int) (otherEndIndex)));
92  if (!currRange.equals(otherRange)) {
93  currRange = currRange.overlap(otherRange);
94  rangeChanged = true;
95  if (currRange == null) {
96  throw createValidationEx(formulaVariable, var, "time");
97  }
98  }
99  }
100  }

102  // if get here then currRange is the range that fits somewhere in all the time axes
103  // now we need to match that up with actual timesteps for each axis and make sure the
104  // date values between that range match for all variables.
105  Date startDate = new Date(currRange.getStart());
106  int timeStart = axes.getTimeStep(startDate);
107  Date endDate = new Date(currRange.getEnd());
108  int timeEnd = axes.getTimeStep(endDate);

113  for (int i = timeStart; i < timeEnd; i++) {
114  tsDates[i] = axes.getDate(i);

116 
117  for (int i = 1; i < variables.size(); i++) {
118  FormulaVariable var = variables.get(i);

120  int otherStart = otherAxes.getTimeStep(startDate);
121  int otherEnd = otherAxes.getTimeStep(endDate);

127  throw new IllegalFormulaException("Time steps are not compatible across datasets");
128  }

148  private void checkTimeRange(List<AxisRange> ranges, Date startDate, Date endDate) throws IllegalFormulaException {
149  AxisRange time = findAxisByType(AxisType.TIME, ranges);
150  if (time == null) return;
151  CoordAxis axis = time.getAxis();
152  // find the dataset the time range axis came from
153  Axes foundAxes = null;
154  for (FormulaVariable var : variables) {
155  Axes axes = var.getDataset().getCoordAxes();
156  if (axis.equals(axes.getTimeAxis())) {
157  foundAxes = axes;
158  break;
159  }
160  }
161 
162  if (foundAxes == null) throw new IllegalFormulaException("Error while resolving time axis");
163  long start = startDate.getTime();
164  long end = endDate.getTime();
165  Range range = time.getRange();
166  GregorianCalendar rangeStartDate = foundAxes.getDate(range.getOrigin());
167  long rangeStart = rangeStartDate.getTimeInMillis();
168  GregorianCalendar rangeEndDate = foundAxes.getDate((int) (range.getOrigin() + (range.getExtent() - 1)));
169  long rangeEnd = rangeEndDate.getTimeInMillis();
170 
171  if (rangeStart >= start && rangeStart <= end && rangeEnd >= start && rangeEnd <= end) {
172  // range is within our "synthetic range" so redo the formula vars to match that
173  for (FormulaVariable var : variables) {
174  Axes<CoordAxis> otherAxes = var.getDataset().getCoordAxes();
175  int tsStart = otherAxes.getTimeStep(rangeStartDate);
176  int tsEnd = otherAxes.getTimeStep(rangeEndDate);
177  // extent so + 1
178  var.setTimeStepRange(new Range(tsStart, (tsEnd - tsStart) + 1));
179  }
180  } else {
181  throw new IllegalFormulaException("Allowable time step range is outside of specified range");
182  }
