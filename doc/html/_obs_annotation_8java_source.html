198  public synchronized void draw(Graphics graphics, int xOffset, int yOffset, int width,

206  final double xMinimum = gridBounds[X][MINIMUM];

208  final double yMinimum = gridBounds[Y][MINIMUM];
209  final double yMaximum = gridBounds[Y][MAXIMUM];
210  final double lonMaximum = (gridCRS == null) ? xMaximum : domain[X][MAXIMUM];
211  final double longitudeShift = lonMaximum > 180.0 ? 360.0 : 0.0;
212  final double xRange = xMaximum - xMinimum;
213  final double yRange = yMaximum - yMinimum;
214  final double xScale = width / xRange;
215  final double yScale = height / yRange;
216 
217  for (ObsData data : list) {
218  double[] point = new double[2];
219  double lon = data.getLon();
220  lon = lon >= 0.0 ? lon : lon + longitudeShift;
221  double lat = data.getLat();
222

225  point[0] = lon;
226  point[1] = lat;
227  }
228  else
229  {// have a projection so project lon/lat to the point
230  point = lonlat2Point(lon, lat, gridCRS);
231  }
232 
233  final int x = (int) Math.round((point[0] - xMinimum) * xScale + xOffset);
234  final int y = (int) Math.round(height + yOffset - (point[1] - yMinimum) * yScale);
235 
236  if (x <= xOffset || x >= xOffset + width)
237  continue;
238 
239  if (y <= yOffset || y >= yOffset + height)
240  continue;
241 
242  final int colorIndex = indexOfObsValue((float) data.getValue(), legendLevels);
243 
244  if (colorIndex == -1)

246 
247  final Color cellColor = legendColors[colorIndex];
248  drawSymbol((Graphics2D) graphics, cellColor, data, x, y);
249  }

356  private double[] lonlat2Point(double lon, double lat, CoordinateReferenceSystem gridCRS)
357  { // convert lon/lat to point
358  double[] aPoint = {0.0, 0.0};
359  try{
360  CRSAuthorityFactory crsFactory = CRS.getAuthorityFactory(true);
361  CoordinateReferenceSystem crs = crsFactory.createCoordinateReferenceSystem("EPSG:4326");
362  boolean lenient = true;
363  MathTransform transform = CRS.findMathTransform(crs, gridCRS, lenient);
364  Coordinate srcCoordinate = new Coordinate(lon, lat);
365  Coordinate targetCoordinate = JTS.transform(srcCoordinate, null, transform);
366  aPoint[0] = targetCoordinate.x;
367  aPoint[1] = targetCoordinate.y;
368  return aPoint;
369  } catch(Exception ex) {
370  Logger.error("Exception converting ObsAnnotation (lon,lat) to projected coordinate:"
371  + ex.toString());
372  Logger.error(ex.getStackTrace());
373  }
374  return null;
375  }
