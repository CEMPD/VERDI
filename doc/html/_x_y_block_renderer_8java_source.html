252  private void updateOffsets() {
253  if (this.blockAnchor.equals(RectangleAnchor.BOTTOM_LEFT)) {

256  } else if (this.blockAnchor.equals(RectangleAnchor.BOTTOM)) {

261  yOffset = 0.0;
262  } else if (this.blockAnchor.equals(RectangleAnchor.LEFT)) {
263  xOffset = 0.0;
264  yOffset = -this.blockHeight / 2.0;
265  } else if (this.blockAnchor.equals(RectangleAnchor.CENTER)) {
266  xOffset = -this.blockWidth / 2.0;

268  } else if (this.blockAnchor.equals(RectangleAnchor.RIGHT)) {
269  xOffset = -this.blockWidth;

271  } else if (this.blockAnchor.equals(RectangleAnchor.TOP_LEFT)) {
272  xOffset = 0.0;

335  public void drawItem(Graphics2D g2, XYItemRendererState state,
336  Rectangle2D dataArea, PlotRenderingInfo info, XYPlot plot,
337  ValueAxis domainAxis, ValueAxis rangeAxis, XYDataset dataset,
338  int series, int item, CrosshairState crosshairState, int pass) {

340  double x = dataset.getXValue(series, item);

344  double min = paintScale.getLowerBound();

346  if (dataset instanceof XYZDataset)

351  z = max;
352 
353  if (z < min)
354  z = min;
355 
356  Color p = (Color)this.paintScale.getPaint(z);
357 
358  double xx0 = domainAxis.valueToJava2D(x + this.xOffset, dataArea,
359  plot.getDomainAxisEdge());
360  double yy0 = rangeAxis.valueToJava2D(y + this.yOffset, dataArea,
361  plot.getRangeAxisEdge());
362  double xx1 = domainAxis.valueToJava2D(x + this.blockWidth
363  + this.xOffset, dataArea, plot.getDomainAxisEdge());
364  double yy1 = rangeAxis.valueToJava2D(y + this.blockHeight
365  + this.yOffset, dataArea, plot.getRangeAxisEdge());
366  Rectangle2D block;
367  PlotOrientation orientation = plot.getOrientation();
368  if (orientation.equals(PlotOrientation.HORIZONTAL)) {
369  block = new Rectangle2D.Double(Math.min(yy0, yy1),
370  Math.min(xx0, xx1), Math.abs(yy1 - yy0),
371  Math.abs(xx0 - xx1));
372  } else {
373  block = new Rectangle2D.Double(Math.min(xx0, xx1),
374  Math.min(yy0, yy1), Math.abs(xx1 - xx0),
375  Math.abs(yy1 - yy0));
376  }
377  g2.setColor(p);
378  g2.fill(block);
379 
380  if (gridLinesEnabled) {
381  boolean aaOn = false;
382  if (g2.getRenderingHint(RenderingHints.KEY_ANTIALIASING) == RenderingHints.VALUE_ANTIALIAS_ON) {
383  g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF);
384  aaOn = true;
385  }
386  g2.setPaint(gridLineColor);
387  g2.setStroke(gridLineStroke);
388  g2.draw(block);
389  if (aaOn) g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
390  } else {
391  g2.setStroke(basicStroke);
392  g2.draw(block);
393  }
394  }
