126  public void writeGrid(String fileName, String gridName, int time, int layer, boolean greyScale, LatLonRect pt)
127  throws IOException {
128  double scaler;
129  GridDataset dataset = ucar.nc2.dt.grid.GridDataset.open(fileName);
130  GridDatatype grid = dataset.findGridDatatype(gridName);
131  GridCoordSystem gcs = grid.getCoordinateSystem();
132 
133  if (grid == null)
134  throw new IllegalArgumentException("No grid named " + gridName + " in fileName");
135  if (!gcs.isRegularSpatial())
136  throw new IllegalArgumentException("Must have 1D x and y axes for " + grid.getName());
137 
138  CoordinateAxis1D xaxis = (CoordinateAxis1D) gcs.getXHorizAxis();
139  CoordinateAxis1D yaxis = (CoordinateAxis1D) gcs.getYHorizAxis();
140  if (!xaxis.isRegular() || !yaxis.isRegular())
141  throw new IllegalArgumentException("Must be evenly spaced grid = " + grid.getName());
142 
143  // read in data
144  Array data = grid.readDataSlice(time, layer, -1, -1);
145  data.getShape();
146  Array lon = xaxis.read();
147  Array lat = yaxis.read();
148 
149  // latlon coord does not need to time 1000.0
150  if (gcs.isLatLon())
151  scaler = 1.0;
152  else
153  scaler = 1000.0;
154 
155  if (yaxis.getCoordValue(0) < yaxis.getCoordValue(1)) {
156  data = data.flip(0);
157  lat = lat.flip(0);
158  }
159 
160  if (gcs.isLatLon()) {
161  data = geoShiftDataAtLon(data, lon);
162  lon = geoShiftLon(lon);
163  }
164  // now it is time to subset the data out of latlonrect
165  // it is assumed that latlonrect pt is in +-180
166  LatLonPointImpl llp0 = pt.getLowerLeftPoint();

168  double minLon = llp0.getLongitude();

170  double maxLon = llpn.getLongitude();
171  double maxLat = llpn.getLatitude();
172 
173  // (x1, y1) is upper left point and (x2, y2) is lower right point
174  int x1 = getLonIndex(lon, minLon, 0);
175  int y1 = getLatIndex(lat, maxLat, 0);
176  int x2 = getLonIndex(lon, maxLon, 1);
177  int y2 = getLatIndex(lat, minLat, 1);
178 
179  // data must go from top to bottom LOOK IS THIS REALLY NEEDED ?
180  double xStart = xaxis.getStart() * scaler;
181  double yStart = Math.abs(yaxis.getMaxValue()) * scaler;
182  double xInc = xaxis.getIncrement() * scaler;
183  double yInc = Math.abs(yaxis.getIncrement()) * scaler;
184 
185  // subsetting data inside the box
186  Array data1 = getYXDataInBox(data, x1, x2, y1, y2);
187  if (pageNumber > 1)
188  geotiff.initTags();
189 
190  // write it out
191  writeGrid(grid, data, greyScale, xStart, yStart, xInc, yInc, pageNumber);
192  pageNumber++;
193

458  private ArrayByte replaceMissingValuesAndScale(GridDatatype grid, Array data) {
459  MAMath.MinMax dataMinMax = grid.getMinMaxSkipMissingData(data);
460  double scale = 254.0 / (dataMinMax.max - dataMinMax.min);
461 
462  ArrayByte byteArray = (ArrayByte) Array.factory(byte.class, data.getShape());
463  IndexIterator dataIter = data.getIndexIterator();
464  IndexIterator resultIter = byteArray.getIndexIterator();
465 
466  byte bv;
467  while (dataIter.hasNext()) {
468  double v = dataIter.getDoubleNext();
469  if (grid.isMissingData(v))
470  bv = 0;
471  else {
472  int iv = (int) ((v - dataMinMax.min) * scale + 1);
473  bv = (byte) (iv & 0xff);
474  }
475  resultIter.setByteNext(bv);
476  }
477 
478  return byteArray;
479  }
