382  private static void computeQuartiles( final float[][][] data,

420  final float firstValue = data[ row ][ column ][ index ];
421  final float lastValue = data[ row ][ column ][ index + last ];
422  final float middleValue = data[ row ][ column ][ index + middle ];
423  final float lowerValue = data[ row ][ column ][ index + lower ];
424  final float upperValue = data[ row ][ column ][ index + upper ];
425 
426  minimum[ row ][ column ] = firstValue;
427  maximum[ row ][ column ] = lastValue;
428  range[ row ][ column ] = lastValue - firstValue;
429 
430  if ( oddNumberOfTimesteps ) {
431  median[ row ][ column ] = middleValue;
432  firstQuartile[ row ][ column ] = lowerValue;
433  thirdQuartile[ row ][ column ] = upperValue;
434  interquartileRange[ row ][ column ] = upperValue - lowerValue;
435  } else {
436  final float averageLowerValue =
437  ( lowerValue + data[ row ][ column ][ index + lower_1 ] ) * 0.5f;
438  final float averageUpperValue =
439  ( upperValue + data[ row ][ column ][ index + upper_1 ] ) * 0.5f;
440 
441  median[ row ][ column ] =
442  ( middleValue + data[ row ][ column ][ index + middle_1]) * 0.5f;
443  firstQuartile[ row ][ column ] = averageLowerValue;
444  thirdQuartile[ row ][ column ] = averageUpperValue;

447 
448  }

834  final double mean ) {
835 
836  final int count = data.length;

838  double s2 = 0.0; // The variance.

842

844  final double value = data[ index ];
845  final double meanDifference = value - mean;

850  sum = newSum;
