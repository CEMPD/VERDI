159  private void processViewGroup(Extension.Parameter param, Perspective perspective) throws PluginDefinitionException {
160  String id = param.getSubParameter(VIEW_GROUP_ID).valueAsString();
161  String parent = getSubParamValueAsString(VIEW_GROUP_PARENT, param);
162  if (parent == null) {
163  if (viewRootFound) {
164  String message = "Multiple view group root groups found";
165  throw new PluginDefinitionException(message);
166  }
167  perspective.createRootGroup(id);
168  viewRootFound = true;
169  } else {
170  String location = getSubParamValueAsString(VIEW_GROUP_LOCATION, param);
171  if (location == null) {
172  String message = "View group '" + id + "' is missing a location parameter";
173  throw new PluginDefinitionException(message);
174  }
175  if (!locationMap.containsKey(location)) {
176  String message = "View group '" + id + "' has an invalid location value; must be one of" +
177  " north, south, east, or west";
178  throw new PluginDefinitionException(message);
179  }
180  float fill = getSubParamValueAsFloat(VIEW_GROUP_FILL, param);
181  // default to .5 if no fill value
182  if (Float.isNaN(fill)) fill = .5f;
183  fill = fill > 1 ? .9f : fill < 0 ? .1f : fill;
184  perspective.createGroup(id, locationMap.get(location), parent, fill);
185  }
